var SedeDao = require("../../app_core/dao/sedeDao");
var Respuesta = require("../helpers/respuesta");

/**
 * Gestiona todo lo referente a los componenetes de una sede
 * @module SedeController
 */

 /**
 *Funcion que permite obtener el listado de departamentos
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAllDepartamentos = function (req, res) {
    SedeDao.findAllDepartamentos().then(function (departamentos) {       
        Respuesta.sendJsonResponse(res, 200, departamentos);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};

 /**
 *Funcion que permite obtener el listado de municipios
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAllMunicipios = function (req, res) {
    SedeDao.findAllMunicipios(req.body).then(function (municipios) {       
        Respuesta.sendJsonResponse(res, 200, municipios);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};

/**
 *Funcion que permite obtener el listado de sedes
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAll = function (req, res) {    
    SedeDao.findAll().then(function (sedes) {
        Respuesta.sendJsonResponse(res, 200, sedes);
    }).catch(function (error) {
        console.log(error);
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};

/**
 *Funcion que permite crear una sede
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var create = function (req, res) {
    var sede = req.body;
    console.log(req.body);
    SedeDao.create(sede).then(function (sede) {
        Respuesta.sendJsonResponse(res, 200, sede);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};

/**
 *Funcion que permite actualizar una sede
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var update = function (req, res) {
    var sede = req.body;
    SedeDao.update(sede).then(function (sede) {
        Respuesta.sendJsonResponse(res, 200, sede);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};

/**
 *Funcion que permite archivar una sede
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var store = function (req, res) {
    var sede = req.body;
    SedeDao.store(sede).then(function (sede) {
        Respuesta.sendJsonResponse(res, 200, sede);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })

/**
 *Funcion que permite eliminar una sede
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var destroy = function (req, res) {
    var sede = req.body;
    SedeDao.destroy(sede).then(function (sede) {
        Respuesta.sendJsonResponse(res, 200, sede);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};
