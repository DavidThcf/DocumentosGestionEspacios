var EspacioFisicoDao = require("../../app_core/dao/espacioFisicoDao");
var Respuesta = require("../helpers/respuesta");

/**
 * Gestiona todo lo referente a los componenetes  de un espacio fisico
 * @module EspacioFisicoController
 */

 /**
 *Funcion que permite obtener el listado de nombres de un espacio fisico
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAllNames = function (req, res) {
    EspacioFisicoDao.findAllNames().then(function (espacioFisico) {
        Respuesta.sendJsonResponse(res, 200, espacioFisico);
    }).catch(function (error) {
        console.log(error);
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};
