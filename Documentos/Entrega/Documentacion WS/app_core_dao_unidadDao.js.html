 var Models = require("../../app_core/models/index");
/**
 * Gestiona todas las consultas referentes a la tabla Unidad
 * @module UnidadDao
 */

 /**
 * Funcion que permite obtener el listado de unidades 
 * @param {json} sede - Objeto con la información necesaria para obtener el listado  de unidades
 */
var findAll = function (sede) {
    return Models.GenerUnidad.findAll({
        where: {
            codigo_localizacion: sede.cod_localizacion,
            estado_unidad: 'A'
        },
        attributes: ['id_unidad', 'nombre_unidad', 'codigo_localizacion'],
        order: [
            ['nombre_unidad']
        ]
    });
};

 /**
 * Funcion que permite obtener el listado de facultades
 */
var findFacultades = function () {
    return Models.UnidadAtributos.findAll({
        include: [{
            model: Models.GenerUnidad,
            where: {
                $or: {
                    id_tipo_unidad: 2,
                    id_unidad: 92
                }
            },
            order: [
                ['id_atributos_espacio_fisico', 'DESC']
            ]

        }]
    });
};

/**
 * Funcion que permite obtener el listado de facultades con datos estadisticos de utilización de aulas de clase
 */
var findFacultadesEstadistica = function () {
    return Models.UnidadAtributos.findAll({
        attributes: ['id_unidad_atr', 'valor'],
        include: [{
            model: Models.GenerUnidad,
            attributes: ['id_unidad', 'nombre_unidad'],
            include: [{
                model: Models.GenerAtributosEspacioFisico, as: 'AtriEspFis',
                attributes: ['id_atributos_espacio_fisico'],
                include: [{
                    model: Models.GenerEspacioFisico
                }],
            }],
            where: {
                $or: {
                    id_tipo_unidad: 2,
                    id_unidad: 92
                }
            },
            order: [
                ['id_atributos_espacio_fisico', 'DESC']
            ]

        }]
    });
};

module.exports.findAll = findAll;
module.exports.findFacultades = findFacultades;
module.exports.findFacultadesEstadistica = findFacultadesEstadistica;
