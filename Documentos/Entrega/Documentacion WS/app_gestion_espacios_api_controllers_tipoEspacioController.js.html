var TipoEspacioDao= require("../../app_core/dao/tipoEspacioDao");
var Respuesta= require("../helpers/respuesta");

/**
 * Gestiona todo lo referente a los componenetes a la disponibilidad de un espacio fisico
 * @module TipoEspacioController
 */

 /**
 *Funcion que permite obtener el listado de (tipo) disponibilidades de un espacio fisico
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAll= function (req,res){
    TipoEspacioDao.findAll().then(function(tipoEspacio){             
        Respuesta.sendJsonResponse(res,200, tipoEspacio);
    }).catch(function(error){
        console.log(error);
        Respuesta.sendJsonResponse(res,500,{"error":'se ha producido un error en la consultaRR'})
    });
};

module.exports.findAll=findAll;
