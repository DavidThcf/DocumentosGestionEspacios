var Express = require('express');
var Multipart = require('connect-multiparty');

/**
 * Administra todas las rutas de servidor del modulo de gestion de espacios fisicos
 * @module GestionEspaciosIndex
 */

 /*
 * @description Variables para el manejo de controladores 
 */
var SedeController = require("../controllers/sedeController");
var AreaConstruidaController = require("../controllers/areaConstruidaController");

var router = Express.Router();
var multipartMiddleware = Multipart();

/***********************************************
 * @description RUTAS PARA LA GESTION DE SEDES 
 ***********************************************/

/**
 * @description Ruta para consultar la lista de departamentos
 */
router.post("/obtener_departamentos",SedeController.findAllDepartamentos);

/**
 * @description Ruta para consultar la lista de municipios
 */
router.post("/obtener_municipios",SedeController.findAllMunicipios);

/**
 * @description Ruta para consultar la lista de sedes
 */
router.post("/obtener_sedes",SedeController.findAll);

/**
 * @description Ruta para crear una sede
 */
router.post("/agregar_sede",SedeController.create);

/**
 * @description Ruta para actualizar una sede
 */
router.put("/actualizar_sede",SedeController.update);

/**
 * @description Ruta para archivar una sede
 */
router.put("/archivar_sede",SedeController.store);

/**
 * @description Ruta para eliminar una sede
 */
router.put("/eliminar_sede",SedeController.destroy);
