var Respuesta = require("../helpers/respuesta");

 /**
 *Funcion que permite obtener el listado de asignaciones de horario a un aula de clase
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAll = function (req, res) {
    AsignacionEspacioDao.findAll(req.body).then(function (asignacionEspacio) {
        Respuesta.sendJsonResponse(res, 200, asignacionEspacio);
    }).catch(function (error) {
        console.log(error);
        Respuesta.sendJsonResponse(res, 500, { "error": 'se ha producido un error en la consulta' })
    });
};
