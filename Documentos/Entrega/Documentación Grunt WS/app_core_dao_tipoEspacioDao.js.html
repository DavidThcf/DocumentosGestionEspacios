var Models = require("../../app_core/models/index");
/**
 * Gestiona todas las consultas referentes a los tipos de espacios físicos (disponibilidad)
 * @module TipoEspacioDao
 */


/**
 * Funcion que permite obtener la lista de disponibilidad de un espacio fisico
 */
var findAll = function () {
    return Models.GenerTipoEspacio.findAll({        
        order: [
            ['nombre']
        ]
    });
};

module.exports.findAll = findAll;
