var Models = require("../../../app_core/models/index");
var sequelize = Models.sequelize;
var Q = require("q");

/**
 * Gestiona todas las consultas referentes a la gestión de espacios físicos
 * @module AsignacionAulaDao
 */

/**
 * Funcion que permite obtener el horario de las aulas de clase
 * @param {Json} asigEsp - Objeto con la informacion nesesaria para obtener el horario de las aulas de clase
 */
var findAll = function (asigEsp) {    
    return Models.HorarioAula.findAll({
        include: [
            {
                model: Models.AsignacionAula,
                required: false,
                separate: true,
                where: {
                    $or: {
                        $and: {
                            id_atributos_espacio_fisico: asigEsp.id_atributos_espacio_fisico,
                            id_dia: asigEsp.id_dia,
                            semestre: asigEsp.semestre,
                            año: new Date().getFullYear() + ""
                        },
                        id_atributos_espacio_fisico: null
                    }
                }
            },
        ],
        order: [
            ['id_horario_aula']
        ]
    });
};

/**
 * Funcion que permite asignar una aula de clase al horario de aulas de clase
 * @param {Json} asigEsp - Objeto con la informacion nesesaria para obtener el horario de las aulas de clase
 */
var transactionCreate = function (asigEsp) {
    var deferred = Q.defer();

    return sequelize.transaction({ autocommit: false }).then(function (t) {
        Models.AsignacionAula.bulkCreate(asigEsp, { transaction: t, returning: true }).then(function (asignaciones) {
            t.commit();
            deferred.resolve(asignaciones);
        }).catch(function (error) {
            t.rollback();
            deferred.reject(error);
        });
        return deferred.promise;
    });
};

/**
 * Funcion que permite eliminar las asignacion de una aula de clase en el horario de aulas de clase
 * @param {Json} hora - Objeto con la informacion nesesaria para eliminar la asignación de un aula en el horario de las aulas de clase
 */
var destroy = function (hora) {
    return Models.AsignacionAula.destroy({
        where: {
            id_asign_aula: hora.id_asign_aula
        }
    });
};

module.exports.findAll = findAll;
module.exports.transactionCreate = transactionCreate;
module.exports.destroy = destroy;
