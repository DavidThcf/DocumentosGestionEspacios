var AreaConstruidaDao= require("../../app_core/dao/areaConstruidaDao");
var Respuesta= require("../helpers/respuesta");

/**
 * Gestiona todo lo referente a los componenetes  de area construida 
 * @module AreaConstruidaController
 */


 /**
 *Funcion que permite obtener un listado de areas construidas
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAll= function (req,res){
    AreaConstruidaDao.findAll(req.body.id_sede).then(function(area){             
        Respuesta.sendJsonResponse(res,200, area);
    }).catch(function(error){
        console.log(error);
        Respuesta.sendJsonResponse(res,500,error);
    });
};


/**
 *Funcion que permite crear un area construida
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var create=function (req,res){   
    AreaConstruidaDao.create(req.body).then(function(area){       
       Respuesta.sendJsonResponse(res,200,area)
    }).catch(function(error){        
        Respuesta.sendJsonResponse(res,500,error);
    });
};

/**
 *Funcion que permite actualizar un area construida
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var update= function (req,res){    
    AreaConstruidaDao.update(req.body).then(function(area){
        Respuesta.sendJsonResponse(res,200,area);
    }).catch(function(error){
        Respuesta.sendJsonResponse(res,500,error);
    });
}

/**
 *Funcion que permite archivar un area construida
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var store = function (req, res) {
    var area = req.body;
    AreaConstruidaDao.store(area).then(function (area) {
        console.log('OK');
        Respuesta.sendJsonResponse(res, 200, area);
    }).catch(function (error) {
        console.log(error);
        Respuesta.sendJsonResponse(res, 500, { "error": error })
    });
};

/**
 *Funcion que permite eliminar un area construida
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var destroy = function (req, res) {
    var area = req.body;
    AreaConstruidaDao.destroy(area).then(function (area) {
        Respuesta.sendJsonResponse(res, 200, area);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, { "error": error })
    });
};


module.exports.findAll=findAll;
module.exports.create=create;
module.exports.update=update;
module.exports.store = store;
module.exports.destroy = destroy;
