var AsignacionAula = require("../../app_core/dao/gestionEspaciosDao/asignacionAulaDao");
var Respuesta = require("../helpers/respuesta");


/**
 * Gestiona todo lo referente a los componenetes  de asignacion  de horas de un aula de clase
 * @module AsignacionAulaController
 */


 /**
 *Funcion que permite obtener el listado de horas asignadas a un aula de clase
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAll = function (req, res) {
    AsignacionAula.findAll(req.body).then(function (asignacionAula) {
        Respuesta.sendJsonResponse(res, 200, asignacionAula);
    }).catch(function (error) {
        console.log(error);
        Respuesta.sendJsonResponse(res, 500, { "error": error })
    });
};

 /**
 *Funcion que permite asignar un horario a las aulas de clase
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var create = function (req, res) {
    asignacion = req.body.asignacion;

    console.log(req.body.facultad);
    var dat = JSON.parse(req.body.facultad);    

    for (var i = 0; i < asignacion.length; i++) {
        asignacion[i].año = new Date().getFullYear();
        asignacion[i].facultad = dat.GenerUnidad.nombre_unidad;
        asignacion[i].color = dat.valor;
        asignacion[i].programa = req.body.programa;
        asignacion[i].materia = req.body.materia;
        asignacion[i].profesor = req.body.profesor;
    }

    AsignacionAula.transactionCreate(asignacion).then(function (asignacionEspacio) {
        Respuesta.sendJsonResponse(res, 200, asignacionEspacio);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 200, error);
    });
};

 /**
 *Funcion que permite eliminar las asignacion de horario de un aula de clase 
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var destroy = function (req, res) {
    AsignacionAula.destroy(req.body).then(function (area) {
        Respuesta.sendJsonResponse(res, 200, area);
    }).catch(function (error) {
        Respuesta.sendJsonResponse(res, 500, error);
    });
}

/*-----------------FIN ASIGNACION AULAS ------------------------------ */

module.exports.findAll= findAll;
module.exports.create = create;
module.exports.destroy = destroy;
