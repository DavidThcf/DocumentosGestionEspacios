var UnidadDao= require("../../app_core/dao/unidadDao");
var Respuesta= require("../helpers/respuesta");

/**
 * Gestiona todo lo referente a los componenetes de las unidades
 * @module UnidadController
 */


 /**
 *Funcion que permite obtener el listado de unidades
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findAll= function (req,res){    
    UnidadDao.findAll(req.body).then(function(unidad){             
        Respuesta.sendJsonResponse(res,200, unidad);
    }).catch(function(error){
        console.log(error);
        Respuesta.sendJsonResponse(res,500,{"error":'se ha producido un error en la consultaRR'})
    });
};

 /**
 *Funcion que permite obtener el listado de facultades
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findFacultades= function (req,res){    
    UnidadDao.findFacultades(req.body).then(function(facultades){
        Respuesta.sendJsonResponse(res,200, facultades);
    }).catch(function(error){
        console.log(error);
        Respuesta.sendJsonResponse(res,500,{"error":error})
    });
};

 /**
 *Funcion que permite obtener el listado de facultades con datos estadisticos de utilización de aulas de clase
 * @param {Object} req - Objeto de peticion de express
 * @param {Object} res - Objeto de respuesta de express
 */
var findFacultadesEstadistica= function (req,res){    
    UnidadDao.findFacultadesEstadistica(req.body).then(function(facultades){         
        Respuesta.sendJsonResponse(res,200, facultades);
    }).catch(function(error){
        console.log(error);
        Respuesta.sendJsonResponse(res,500,{"error":error})
    });
};



module.exports.findAll=findAll;
module.exports.findFacultades=findFacultades;
module.exports.findFacultadesEstadistica=findFacultadesEstadistica;
