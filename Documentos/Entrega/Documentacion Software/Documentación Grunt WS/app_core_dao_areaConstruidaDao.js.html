var Models = require("../../app_core/models/index");

/**
 * Gestiona todas las consultas referentes a la gestión áreas construidas
 * @module AreaConstruidaDao
 */


/**
 * Funcion que permite obtener el listado de areas construidas
 * @param {integer} id_sede - identificador unico de las sedes
 */
var findAll = function (id_sede) {
    return Models.GenerAreaConstruida.findAll({
        where: {
            id_sede: id_sede,
            estado: 'A'
        },
        order: [
            ['nombre']
        ]
    });
};

/**
 * Funcion que permite crear un area contruida
 * @param {json} areaConstruida - Objeto con la informacion necesaria para crear un area construida
 */
var create = function (areaConstruida) {
    return Models.GenerAreaConstruida.create({
        nombre: areaConstruida.nombre,
        id_sede: areaConstruida.id_sede,
        num_pisos: areaConstruida.num_pisos,
        mts2: areaConstruida.mts2,
        mts2_acad: areaConstruida.mts2_acad,
        mts2_no_acad: areaConstruida.mts2_no_acad
    });
};

/**
 * Funcion que permite actualizar un area contruida
 * @param {json} areaConstruida - Objeto con la informacion necesaria para actualizar un area construida
 */
var update = function (areaConstruida) {
    return Models.GenerAreaConstruida.update(
        {
            nombre: areaConstruida.nombre,
            capacidad: 0,
            id_sede: areaConstruida.id_sede,
            num_pisos: areaConstruida.num_pisos,
            mts2_acad: areaConstruida.mts2_acad,
            mts2_no_acad: areaConstruida.mts2_no_acad,
            mts2: areaConstruida.mts2
        },
        {
            where: { id_area_construida: areaConstruida.id_area_construida }
        }
    );
};

/**
 * Funcion que permite archivar un area contruida
 * @param {json} areaConstruida - Objeto con la informacion necesaria para archivar un area construida
 */
var store = function (areaConstruida) {
    return Models.GenerAreaConstruida.update(
        {
            estado: 'I'
        },
        {
            where: { id_area_construida: areaConstruida.id_area_construida }
        }
    );
};

/**
 * Funcion que permite eliminar un area contruida
 * @param {json} areaConstruida - Objeto con la informacion necesaria para eliminar un area construida
 */
var destroy = function (areaConstruida) {
    return Models.GenerAreaConstruida.destroy({
        where: {
            id_area_construida: areaConstruida.id_area_construida
        }
    });
};


module.exports.findAll = findAll;
module.exports.create = create;
module.exports.update = update;
module.exports.store = store;
module.exports.destroy = destroy;
